wols = function(X,Y,W,Vg=NULL){
  n = nrow(X)
  K = ncol(X)
  ni = ncol(Y)
  beta_hat = matrix(nrow=ni,ncol=K)
  for(j in 1:ni){
    Xw = X*sqrt(W[,j])
    yw = Y[,j]*sqrt(W[,j])
    
    A = t(Xw)%*%Xw
    if(!is.null(Vg)){
      Vw = Vg*(W[,j])
      V = diag(c(colSums(Vw)))
    }else{
      V = 0
    }
    
    bhat = solve(A-V)%*%t(Xw)%*%yw
    bhat = pmax(bhat,0)
    bhat = bhat/sum(bhat)
    beta_hat[j,] = bhat
  }
  beta_hat
}
